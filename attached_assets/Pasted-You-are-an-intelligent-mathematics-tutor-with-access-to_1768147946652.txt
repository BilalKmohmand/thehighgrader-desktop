def extract_topic_keywords(problem):
    """Extract main math topics from problem"""
    # Use NLP or keyword matching
    keywords = []
    topics = {
        'derivative': ['derivative', 'differentiate', "d/dx", "f'(x)"],
        'integral': ['integrate', 'integral', '∫'],
        'limit': ['limit', 'lim', 'approaches'],
        'solve': ['solve', 'find x', 'equation'],
        # Add more...
    }
    
    for topic, terms in topics.items():
        if any(term in problem.lower() for term in terms):
            keywords.append(topic)
    
    return ' '.join(keywords)

def deduplicate_by_similarity(chunks, threshold=0.9):
    """Remove very similar chunks"""
    unique = []
    for chunk in chunks:
        is_duplicate = False
        for existing in unique:
            if calculate_similarity(chunk['content'], existing['content']) > threshold:
                is_duplicate = True
                break
        if not is_duplicate:
            unique.append(chunk)
    return unique
